# 【组件】 卡片模板组件 B

# 描述

标题 + 副标题

# 使用效果

# 使用方法

在.ux文件中引入组件

<import
  name="template-b"
  src="vivo-cards-suits/components/voice/components/template-b.ux"
></import>

# 示例

<template>
  <div class="card">
    <template-b title="{{title}}" subtitle="{{subtitle}}"></template-b>
  </div>
</template>

<script>
  export default {
    data() {
      return {
        title: "标题",
        subtitle: "只需要输入文字，列表高度将自动匹配。"
      };
    }
  };
</script>

<style lang="less">
  .card {
    width: 100%;
    flex-direction: column;
  }
</style>
