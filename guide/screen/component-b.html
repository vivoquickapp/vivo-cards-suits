# 【组件】列表组件 B

# 描述

列表样式，用于图文列表

# 使用效果

# 使用方法

在.ux文件中引入组件

<import
  name="list-item-b"
  src="vivo-cards-suits/components/screen/components/list-item-b"
></import>

# 组件属性

属性 类型 默认值 说明
title string - 标题
subTitle string - 副标题
button object - 按钮对象
button.text string '按钮' 按钮文本
button.url string - 按钮跳转地址
